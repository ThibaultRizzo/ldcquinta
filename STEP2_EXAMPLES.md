// How the progress bar is calculated
function updateProgress() {
    let totalFields = 0;
    let filledFields = 0;
    
    selectedServices.forEach(service => {
        totalFields += service.fields.length;
        
        service.fields.forEach(field => {
            const fieldId = `${service.id}-${field.name}`;
            const element = document.getElementById(fieldId);
            
            if (element && element.value && 
                element.value !== 'Select...' && 
                element.value !== 'SÃ©lectionner...') {
                filledFields++;
            }
        });
    });
    
    // Update UI
    const remaining = totalFields - filledFields;
    const progressPercent = (filledFields / totalFields) * 100;
    
    document.getElementById('remainingFields').textContent = remaining;
    document.getElementById('progressFill').style.width = progressPercent + '%';
}

// How checkmarks are shown when service is complete
function updateSidebarStatus() {
    selectedServices.forEach((service, index) => {
        const item = document.querySelectorAll('.service-item')[index];
        let allRequiredFieldsFilled = true;
        
        service.fields.forEach(field => {
            if (!field.required) return;
            
            const fieldId = `${service.id}-${field.name}`;
            const element = document.getElementById(fieldId);
            
            if (!element || !element.value) {
                allRequiredFieldsFilled = false;
            }
        });
        
        // Add or remove checkmark
        if (allRequiredFieldsFilled) {
            item.classList.add('completed'); // Shows âœ“
        } else {
            item.classList.remove('completed');
        }
    });
}
